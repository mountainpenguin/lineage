[docs]class Cell(object):
    def __init__(self, cell):
        for d in range(len(cell.dtype)):
            self.__dict__[cell.dtype.names[d]] = cell[d]




[docs]class Frames(object):
    def __init__(self, frames):
        self.frames = frames
        self._idx = {}
        for f in self.frames:
            for id in f._idx.keys():
                self._idx[id] = f.frame


[docs]    def cell(self, id):
        if id in self._idx:
            return self.frames[self._idx[id]].cell(id)




[docs]class Frame(object):
    def __init__(self, frame, cells):
        self.frame = frame
        self.cells = cells
        # make cell id index
        self._idx = {}
        for c in self.cells:
            self._idx[c.id] = c.py_idx


[docs]    def cell(self, id):
        if id in self._idx:
            return self.cells[self._idx[id]]




[docs]def main():
    print("Loading alignment data...")
    alignment = scipy.io.loadmat("mt/alignment.mat")
    print(">>> loaded")
    alignment = alignment["shiftframes"]
    alignment_x = alignment[0][0][0][0]
    alignment_y = alignment[0][0][1][0]
    alignment = np.array([alignment_x, alignment_y]).T
    print(">>> parsed")

    print("Loading cell data...")
    cells = scipy.io.loadmat("mt/mt.mat")
    print(">>> loaded")
    cellList = cells["cellList"][0]

    cell_total = 0
    pos_idx = 0
    frames = []

    if os.path.exists("uuids.json"):
        uuids = json.loads(open("uuids.json").read())
    else:
        uuids = {}
        # "frame:mt_idx": uuid

    if os.path.exists("lineages.json"):
        lineages = json.loads(open("lineages.json").read())
    else:
        lineages = {}
        # uuid: parent, child

    for pos in cellList:
        if pos.shape[0] == 0:
            pos_idx += 1
            continue
        cell_idx = 0
        py_cell_idx = 0
        frame_cells = []
        for c in pos[0]:
            if c.shape != (0, 0):
                cell = Cell(c[0][0])
                if "length" not in cell.__dict__:
                    input("Please reselect Cell {0} in Frame {1}".format(
                        cell_idx, pos_idx
                    ))
                else:
                    cell.frame = pos_idx + 1
                    cell.mt_idx = cell_idx + 1
                    cell.py_idx = py_cell_idx
                    uuid_chk = "frame:{0}:{1}".format(cell.frame, cell.mt_idx)
                    if uuid_chk in uuids:
                        cell.id = uuids[uuid_chk]
                    else:
                        cell.id = uuid.uuid4().urn[9:]
                        uuids[uuid_chk] = cell.id

                    # check for lineages
                    if cell.id in lineages:
                        cell.parent = lineages[cell.id][0]
                        cell.children = lineages[cell.id][1]
                    else:
                        cell.parent = None
                        cell.children = None

                    py_cell_idx += 1
                    cell_total += 1
                    frame_cells.append(cell)
            cell_idx += 1
        frame = Frame(pos_idx, frame_cells)
        frames.append(frame)
        pos_idx += 1
    print(">>> parsed")
    print(">>> {0} frames, {1} cells".format(
        pos_idx, cell_total
    ))

    files = sorted(glob.glob("*.tif"))

    # desired lineage
    #lin1 = "ef49d994-d204-4d37-9962-1c32724e0195"
    #lin1 = "a20dccbc-dd7a-494d-9bd7-4e223bf3291a"
    lin1 = "ac8d7868-9c64-4778-ab83-6398ec5e0964"
    frames_obj = Frames(frames)

    lin = []
    cell = frames_obj.cell(lin1)
    if not os.path.exists("movie-frames"):
        os.mkdir("movie-frames")

    lineage = []
    spare_children = []
    while True:
        uid = cell.id
        length = cell.length[0][0]
        lineage.append(
            (uid, length, cell.frame)
        )
        if type(cell.children) is str:
            cell = frames_obj.cell(cell.children)
        elif type(cell.children) is list:
            spare_children.append(frames_obj.cell(cell.children[1]))
            cell = frames_obj.cell(cell.children[0])
            lin.append(lineage)
            lineage = []
        else:
            lin.append(lineage)
            lineage = []

            if spare_children:
                cell = spare_children.pop(0)
            else:
                break

    cell = frames_obj.cell(lin1)
    c_x = np.mean([
        cell.mesh[:, 0], cell.mesh[:, 2]
    ])
    c_y = np.mean([
        cell.mesh[:, 1], cell.mesh[:, 3]
    ])
    bounds_master = [
        c_x - 160, c_y - 160,
        c_x + 160, c_y + 160
    ]

    frame_num = 1
    for field_file in files:
        print("Handling frame {0}...".format(frame_num))
        field_img = scipy.misc.imread(field_file)

        field_img = field_img - field_img.min()
        # scale to 255
        field_img = (field_img / (field_img.max())) * 255
        field_img = np.uint8(field_img)
        field_image = PIL.Image.fromstring(
            #"I;16",
            "L",
            (field_img.shape[1], field_img.shape[0]),
            field_img.tostring()
        )
        field_image = field_image.convert("RGB")
        print(">>> loaded full frame")

        print(">>> reading offset")
        offset = alignment[frame_num - 1]
        bounds = list(bounds_master)
        bounds[0] -= offset[1]
        bounds[1] -= offset[0]
        bounds[2] -= offset[1]
        bounds[3] -= offset[0]

        draw = PIL.ImageDraw.Draw(field_image)
        draw.rectangle(bounds, outline=(255, 255, 255))
        print(">>> rectangle drawn")

        field_image = field_image.resize((672, 512), PIL.Image.ANTIALIAS)
        print(">>> frame resized")

        crop_img = field_img[
            bounds[1]:bounds[3],
            bounds[0]:bounds[2],
        ]
        crop_image = PIL.Image.fromstring(
            "L",
            (crop_img.shape[1], crop_img.shape[0]),
            crop_img.tostring()
        )
        crop_image = crop_image.convert("RGB")
        print(">>> frame cropped")

        dimensions = (
            field_image.size[0],
            field_image.size[1] + 100 + crop_image.size[1]
        )
        whole_image = PIL.Image.new(
            "RGB",
            dimensions,
            color=(255, 255, 255),
        )
        whole_image.paste(
            field_image, (0, 0)
        )
        whole_image.paste(
            crop_image,
            (0,
             field_image.size[1] + 50)
        )

        colours = ["b.-", "r.-", "g.-", "c.-", "m.-", "y.-", "k.-",
                   "b^-", "r^-", "g^-", "c^-", "m^-", "y^-", "k^-",
                   "b--", "r--", "g--", "c--", "m--", "y--", "k--"]
        plt.figure(figsize=(3.62, 2.79), dpi=100)
        plt.xlim([0, 83])
        plt.ylim([0, 120 * 0.077])
        for l in lin:
            c = colours.pop(0)
            d = list(filter(
                lambda x: x[2] <= frame_num,
                l
            ))
            if d:
                ll = np.array([(x[2], x[1] * 0.077) for x in d])
                plt.plot(ll[:, 0], ll[:, 1], c)

        # draw RIF rectangle
        rect = matplotlib.patches.Rectangle(
            (32, 0),
            28,
            130 * 0.077,
            facecolor="y",
            edgecolor=None,
            alpha=.3,
        )

        ax = plt.gca()
        ax.add_patch(rect)

        canvas = plt.get_current_fig_manager().canvas
        canvas.draw()
        plot_wh = canvas.get_width_height()
        plot_image = PIL.Image.fromstring(
            "RGB",
            plot_wh,
            canvas.tostring_rgb()
        )
        print(">>> lineages plotted")

        # crop_image: 310, 310
        # field_image: 672, 512
        # whole_image: 672, 812
        # plot_image: 372, 279
        whole_image.paste(
            plot_image,
            (320, 570)
        )

        if frame_num >= 32 and frame_num <= 60:
            font = PIL.ImageFont.truetype(
                ("/usr/share/texmf-dist/fonts/truetype/public"
                 "/dejavu/DejaVuSans-Bold.ttf"),
                24
            )
            draw = PIL.ImageDraw.Draw(whole_image)
            draw.text(
                (0, 0),
                "+ RIF",
                (255, 0, 0),
                font=font
            )

        whole_image.save(
            "movie-frames/{0:05d}.png".format(frame_num)
        )
        plt.close()

        frame_num += 1
        print("done")


if __name__ == "__main__":
    main()
